`InfoAlert` is a modal panel. It contains no modal logic, but can be used together with `useAlertDialog`, `useModalDialog`, etc.

```tsx
const SessionExpiredAlert: React.FC = () => {
  const { resolve } = useDialogPromise();

  return (
    <InfoAlert
      maxWidth={"384px"}
      onRequestClose={resolve}
      heading={heading}
      text={text}
      icon={stenaBell}
      buttons={<PrimaryButton size={"larger"} label={"Restart"} />}
    />
  );
};

const Example: React.FC = () => {
  const [element, { show }] = useAlertDialog(SessionExpiredAlert);

  return (
    <Row>
      <PrimaryButton label={"Open InfoAlert"} onClick={() => show()} />
      {element}
    </Row>
  );
};
```
